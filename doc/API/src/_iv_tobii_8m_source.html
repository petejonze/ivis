302  */
303  itime = double(timestamp)/1000000; /* microseconds to seconds (since GetSecs is in seconds)
304  *offset = itime - timeReceived;
305  *t = itime - offset;
306  */
307  CPUtime = itime - (itime(end) - timeReceived); /* effectively just substituting the current time for the last sampled time
308  */
309 
310  /* extract relevant gaze data
311  */
312  xy1 = lefteye(:,7:8); /* see ParseGazeData

314  xy2 = righteye(:,7:8);
315  /* additional
316  */
317  p1 = lefteye(:,12); /* pupil diameter
318  */
319  v1 = lefteye(:,13); /* validity
320  */
321  p2 = righteye(:,12);
322  v2 = righteye(:,13);
323 
324  /* exclude any erroneous points (which Tobii marks as -1(?)), by
325  * setting to NaN
326  */
327  xy1(xy1==-1) = NaN;
328  xy2(xy2==-1) = NaN;
329 
330  /* combine data across eyes (or select from one only)
331  */
332  switch obj.eyes
333  case 0 /* mean of both
334  */
335  xy = nanmean(cat(3,xy1,xy2),3);
336  vldty = v1*10 + v2;
337  pd = nanmean(cat(3,p1,p2),3);
338  case 1 /* left eye only
339  */
340  xy = xy1;
341  vldty = v1;
342  pd = p1;
343  case 2 /* right eye only
344  */
345  xy = xy2;
346  vldty = v2;
347  pd = p2;

350  end
